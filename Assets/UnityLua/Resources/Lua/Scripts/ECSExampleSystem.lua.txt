local viewComponents = nil

function initialize()
    print("initialize", self)
    viewComponents = system.Create(self, CS.UniEasy.ECS.ViewComponent)
end

function onenable()
    print("onenable")
    subscribe(
        group.OnAdd(viewComponents),
        function(entity)
            local viewComponent = get_component(entity, CS.UniEasy.ECS.ViewComponent)
            print(viewComponent.Transforms[0].name)
        end
    )
end

function ondisable()
    print("ondisbale")
end

function ondestroy()
    print("ondestroy")
end
