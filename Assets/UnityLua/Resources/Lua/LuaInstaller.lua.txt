function installbindings()
    print("installbindings")
    local globalSystems = CS.UnityEngine.Resources.Load("Lua/GlobalSystems")
    local paths = CS.UniEasy.StringExtensions.Split(globalSystems.text, "\n")
    for i = 0, paths.Length - 1 do
        require(paths[i])
    end
end
